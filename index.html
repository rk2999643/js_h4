 return min;

Q.11 Birthday Game 

         
function Birthday_Game(arr,D ,M) {
  let count=0;
  for(i=0;arr.length-M+1>i;i++){
    let sum=0;
    for(let j=i;(M+i)>i;j++){
      sum+=arr[j];
    }
    if(sum===D){
      count++;
    }
  }
  return count;
 
 }
